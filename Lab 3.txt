#include <iostream>
using namespace std;
void readScores(int arr[], int n);
void assignHouses(int arr[], int n);

int main() {
    int matrix[5] = {10, 20, 30, 40, 50};
    
    // Task 1
    int sum = 0;
    for (int i = 0; i < 5; i++) {
        sum += matrix[i];
    }
    cout << "Sum = " << sum << endl;
    
    // Task 2
    for (int i = 4; i > -1; i--) {
        cout << matrix[i] << endl;
    }
    
    // Task 3
    int n;
    cout << "Enter Number Of Students: ";
    cin >> n;
    
    int arr[n];
    readScores(arr, n);
    assignHouses(arr, n);
    
    return 0;
}

void readScores(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << "Enter Score: ";
        cin >> arr[i]; 
    }
}
    
void assignHouses(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << "Student " << i + 1 << ", ";  
        if (arr[i] >= 25 && arr[i] < 40) {
            cout << "House: Slytherin" << endl;
        }
        else if (arr[i] >= 40 && arr[i] < 55) {
            cout << "House: Hufflepuff" << endl;
        }
        else if (arr[i] >= 55 && arr[i] < 70) {
            cout << "House: Ravenclaw" << endl;
        }
        else if (arr[i] >= 70) {
            cout << "House: Gryffindor" << endl;
        }
        else {
            cout << "House: Not Assigned (score too low)" << endl;  
        }
    }
}
